 Welcome to Portal 2D!

 "H" - hero
 "." - aim
 "O" - portal
 "E" - exit
 "X" - impassable wall
 "`" - button to turn off impassable wall
 "=" - patrol_turret
 "+" - turret_hunter

 Controls:
 moving - a, d
 jump - spacebar
 aiming - arrows "up, down, left, right"
 set blue portal - e
 set red portal - q
 action - enter
 pause - p
 quit the level - backspace
